# 直线到直线的距离

计算直线到直线的距离

import * as cga from "xtorcga";
var line0 = new cga.Line(randomV3(), randomV3());
var line1 = new cga.Line(randomV3(), randomV3());
var result = line0.distanceLine(line1);
